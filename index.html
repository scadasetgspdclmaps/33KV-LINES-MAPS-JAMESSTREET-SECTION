// ══════════════════════════════════════════════════ // 1. MAP & BASEMAP INIT // ══════════════════════════════════════════════════ const map = L.map('map', { center: [17.4437, 78.4785], zoom: 16, zoomControl: false, preferCanvas: true }); const BM = { dark: L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', { maxZoom: 22 }), satellite: L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', { maxZoom: 22 }), street: L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { maxZoom: 22 }), hybrid: L.tileLayer('https://mt1.google.com/vt/lyrs=y&x={x}&y={y}&z={z}', { maxZoom: 22 }) }; let curBM = 'dark'; BM.dark.addTo(map); // ══════════════════════════════════════════════════ // 2. LAYER STATE & COLLECTIONS // ══════════════════════════════════════════════════ const layerGroups = { 'ug-lines': L.layerGroup().addTo(map), 'oh-lines': L.layerGroup().addTo(map), 'points': L.layerGroup().addTo(map), 'ab': L.layerGroup().addTo(map), 'dp': L.layerGroup().addTo(map), 'cut': L.layerGroup().addTo(map), 'ss33': L.layerGroup().addTo(map), 'ehts': L.layerGroup().addTo(map) }; const allFeatures = []; // For Search const feederLayers = {}; // To isolate by feeder name // ══════════════════════════════════════════════════ // 3. GIS ENGINE // ══════════════════════════════════════════════════ function getMarkerStyle(type) { const base = { radius: 5, fillOpacity: 1, weight: 1.5, color: '#ffffff' }; const colors = { ab: '#ff9800', dp: '#9c27b0', points: '#00c8ff', cut: '#78909c', ss33: '#00e676', ehts: '#ff1744' }; return { ...base, fillColor: colors[type] || '#fff' }; } function initGIS() { let stats = { total: 0, ug: 0, oh: 0, ab: 0, dp: 0, cp: 0 }; const uniqueFeeders = new Set(); DATA_REGISTRY.forEach(item => { const data = window[item.varName]; if (!data) return; uniqueFeeders.add(item.feeder); if (!feederLayers[item.feeder]) feederLayers[item.feeder] = []; const leafletLayer = L.geoJSON(data, { style: (f) => ({ color: item.cable === 'UG' ? '#2196f3' : '#f44336', weight: 3, opacity: 0.8, dashArray: item.cable === 'OH' ? '5, 8' : '' }), pointToLayer: (f, ll) => L.circleMarker(ll, getMarkerStyle(item.type)), onEachFeature: (f, layer) => { // Add to search index allFeatures.push({ feature: f, layer: layer, feeder: item.feeder }); // Popup Content const p = f.properties; layer.bindPopup(`
${item.feeder}
${p.Name || p.POLE_NO || '33KV Equipment'}
Type${item.type.toUpperCase()}
Cable${item.cable}
`); } }); // Grouping logic const target = (item.type === 'lines') ? (item.cable === 'UG' ? 'ug-lines' : 'oh-lines') : item.type; if (layerGroups[target]) leafletLayer.addTo(layerGroups[target]); feederLayers[item.feeder].push(leafletLayer); // Stats accumulation stats.total += data.features.length; if (item.type === 'ab') stats.ab += data.features.length; if (item.type === 'dp') stats.dp += data.features.length; if (item.type === 'points') stats.cp += data.features.length; if (item.type === 'lines') { if (item.cable === 'UG') stats.ug++; else stats.oh++; } }); populateFeederFilters(Array.from(uniqueFeeders)); updateUI(stats); hideLoader(); } // ══════════════════════════════════════════════════ // 4. UI & FILTERS // ══════════════════════════════════════════════════ function populateFeederFilters(feeders) { const container = document.getElementById('feederChecks'); feeders.sort().forEach(name => { const div = document.createElement('div'); div.className = 'lrow'; div.innerHTML = `
${name}
`; div.onclick = () => toggleFeederIsolation(name, div); container.appendChild(div); }); } function toggleFeederIsolation(name, element) { const isOff = element.classList.toggle('off'); feederLayers[name].forEach(lyr => { if (isOff) map.removeLayer(lyr); else map.addLayer(lyr); }); toast(isOff ? `Hidden: ${name}` : `Showing: ${name}`); } function updateUI(s) { document.getElementById('st-total').innerText = s.total; document.getElementById('st-ug').innerText = s.ug; document.getElementById('st-oh').innerText = s.oh; document.getElementById('st-ab').innerText = s.ab; document.getElementById('st-dp').innerText = s.dp; document.getElementById('st-cp').innerText = s.cp; Object.keys(layerGroups).forEach(k => { const el = document.getElementById('cnt-' + k); if (el) el.innerText = layerGroups[k].getLayers().length; }); } // ══════════════════════════════════════════════════ // 5. SEARCH & UTILS // ══════════════════════════════════════════════════ document.getElementById('srchInput').addEventListener('keypress', function(e) { if (e.key === 'Enter') { const term = this.value.toLowerCase(); const found = allFeatures.find(item => { const p = item.feature.properties; return (p.Name && p.Name.toLowerCase().includes(term)) || (p.POLE_NO && p.POLE_NO.toLowerCase().includes(term)); }); if (found) { const coords = found.layer.getBounds ? found.layer.getBounds().getCenter() : found.layer.getLatLng(); map.flyTo(coords, 20); found.layer.openPopup(); toast(`Found in ${found.feeder}`); } else { toast("No asset found with that ID"); } } }); function setBM(n) { map.removeLayer(BM[curBM]); BM[n].addTo(map); curBM = n; document.querySelectorAll('.bmbtn').forEach(b => b.classList.remove('active')); document.getElementById('bm-' + n).classList.add('active'); } function toggleLyr(k) { const l = layerGroups[k], row = document.getElementById('lyr-' + k); if (map.hasLayer(l)) { map.removeLayer(l); row.classList.add('off'); } else { map.addLayer(l); row.classList.remove('off'); } } function togglePanel() { const p = document.getElementById('panel'), t = document.getElementById('ptab'), l = document.getElementById('legend'); p.classList.toggle('off'); t.classList.toggle('off'); l.classList.toggle('shift'); t.innerText = p.classList.contains('off') ? '▶' : '◀'; setTimeout(() => map.invalidateSize(), 400); } function toast(m) { const t = document.getElementById('toast'); t.innerText = m; t.classList.add('show'); setTimeout(() => t.classList.remove('show'), 3000); } function hideLoader() { document.getElementById('ldfill').style.width = '100%'; setTimeout(() => document.getElementById('loader').classList.add('gone'), 600); } function simStatus(type) { const chip = document.getElementById('chipStatus'); chip.className = `chip chip-${type}`; chip.innerText = `● ${type.toUpperCase()}`; toast(`System simulation: ${type.toUpperCase()}`); } window.addEventListener('load', initGIS); map.on('mousemove', e => { document.getElementById('coords').innerText = `${e.latlng.lat.toFixed(5)}, ${e.latlng.lng.toFixed(5)}`; });
